// ========== 업로드 진단 함수 ==========
function runDiagnose() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var ui = SpreadsheetApp.getUi();
  var results = [];

  // 1. 검증결과 시트 상태
  var vs = ss.getSheetByName('검증결과');
  if (!vs) {
    results.push('❌ 검증결과 시트가 없습니다.');
  } else {
    var rows = vs.getLastRow() - 1;
    results.push('✅ 검증결과 시트 있음: ' + rows + '행');
    var header = vs.getRange(1, 1, 1, vs.getLastColumn()).getValues()[0];
    var finalCol = -1;
    for (var h = 0; h < header.length; h++) {
      if (String(header[h]).indexOf('최종판정') >= 0) finalCol = h;
    }
    results.push(finalCol >= 0 ? '✅ 최종판정 컬럼 위치: ' + finalCol : '❌ 최종판정 컬럼 없음');
    if (rows > 0 && finalCol >= 0) {
      var data = vs.getDataRange().getValues();
      var p = 0, c = 0;
      for (var i = 1; i < data.length; i++) {
        var v = String(data[i][finalCol] || '').trim();
        if (v === 'PASS') p++;
        else if (v === 'CORRECTED') c++;
      }
      results.push('📊 PASS: ' + p + ', CORRECTED: ' + c + ' = 총 ' + (p + c) + '개 업로드 대상');
    }
  }

  // 2. 서비스 계정 상태
  try {
    var raw = PropertiesService.getScriptProperties().getProperty('FIREBASE_SERVICE_ACCOUNT');
    if (!raw) {
      results.push('❌ FIREBASE_SERVICE_ACCOUNT 속성이 없습니다.');
    } else {
      var k = JSON.parse(raw);
      results.push('✅ 서비스 계정 있음: ' + k.client_email);
    }
  } catch (e) {
    results.push('❌ 서비스 계정 JSON 파싱 실패: ' + e.message);
  }

  // 3. 토큰 발급 테스트
  try {
    var tok = getToken_();
    if (tok) {
      results.push('✅ JWT 토큰 발급 성공 (길이: ' + tok.length + ')');
      // Firestore 연결 테스트
      var testUrl = 'https://firestore.googleapis.com/v1/projects/' + PROJECT_ID + '/databases/(default)/documents/test_conn';
      var r2 = UrlFetchApp.fetch(testUrl, {
        method: 'get',
        headers: { 'Authorization': 'Bearer ' + tok },
        muteHttpExceptions: true
      });
      var code = r2.getResponseCode();
      results.push('🔗 Firestore Bearer 테스트: ' + code);
      if (code !== 200) results.push('   응답: ' + r2.getContentText().substring(0, 200));
    } else {
      results.push('❌ 토큰 발급 실패 (null 반환)');
    }
  } catch (e) {
    results.push('❌ 토큰/연결 테스트 오류: ' + e.message);
  }

  ui.alert('🔍 업로드 진단 결과\n\n' + results.join('\n'));
}

// ★★★★★ 강화된 parseJSON_ 함수 ★★★★★
function parseJSON_(text) {
  if (!text) return null;
  var cleaned = String(text)
    .replace(/```json\n?/gi, '').replace(/```\n?/g, '').trim()
    .replace(/[\u201C\u201D]/g, '"').replace(/[\u2018\u2019]/g, "'")
    .replace(/""/g, '"');
  var start = -1, firstBracket = cleaned.indexOf('['), firstBrace = cleaned.indexOf('{');
  if (firstBracket >= 0 && (firstBrace < 0 || firstBracket < firstBrace)) start = firstBracket;
  else if (firstBrace >= 0) start = firstBrace;
  if (start < 0) return null;
  var stack = [], inString = false, escape = false, end = start;
  for (var i = start; i < cleaned.length; i++) {
    var ch = cleaned[i];
    if (escape) { escape = false; continue; }
    if (ch === '\\') { escape = true; continue; }
    if (ch === '"') { inString = !inString; continue; }
    if (!inString) {
      if (ch === '[' || ch === '{') stack.push(ch);
      else if (ch === ']') { if (stack.length && stack[stack.length-1] === '[') stack.pop(); if (stack.length === 0) { end = i; break; } }
      else if (ch === '}') { if (stack.length && stack[stack.length-1] === '{') stack.pop(); if (stack.length === 0) { end = i; break; } }
    }
  }
  if (end === start) return null;
  var jsonStr = cleaned.substring(start, end + 1);
  jsonStr = jsonStr.replace(/[\n\r\t]/g, ' ').replace(/\s+/g, ' ').replace(/,\s*}/g, '}').replace(/,\s*]/g, ']');
  jsonStr = jsonStr.replace(/([{,]\s*)([a-zA-Z0-9_]+)(\s*:)/g, '$1"$2"$3');
  try { return JSON.parse(jsonStr); } catch(e) {
    try { return JSON.parse(jsonStr.replace(/(\w+):/g, '"$1":')); } catch(e2) { log_('parseJSON_', '❌ 파싱 실패: ' + e2.message + '\n원본: ' + jsonStr.substring(0,300)); return null; }
  }
}

function callGemini_(prompt, temp, maxTokens) {
  var options = { method: 'post', contentType: 'application/json', payload: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { temperature: temp||0.2, maxOutputTokens: maxTokens||8000 } }), muteHttpExceptions: true, timeout: 60000 };
  var res = UrlFetchApp.fetch('https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key='+GEMINI_KEY, options);
  var json = JSON.parse(res.getContentText());
  if (json.error) throw new Error('Gemini: '+json.error.message);
  return json.candidates[0].content.parts[0].text;
}

function callDeepSeek_(prompt, maxTokens) {
  var options = {
    method: 'post',
    headers: { 'Authorization': 'Bearer ' + DEEPSEEK_KEY, 'Content-Type': 'application/json' },
    payload: JSON.stringify({
      model: 'deepseek-chat',
      messages: [{ role: 'user', content: prompt }],
      max_tokens: maxTokens || 8000,
      temperature: 0.6,
      response_format: { type: "json_object" }
    }),
    muteHttpExceptions: true,
    timeout: 120000
  };
  var res = UrlFetchApp.fetch('https://api.deepseek.com/v1/chat/completions', options);
  var json = JSON.parse(res.getContentText());
  if (json.error) throw new Error('DeepSeek: ' + json.error.message);
  return json.choices[0].message.content;
}

function getUnitKeys_(filter) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var refSheet = ss.getSheetByName('참고문제');
  if (!refSheet || refSheet.getLastRow()<=1) return { keys: [], byUnit: {} };
  var refData = refSheet.getDataRange().getValues();
  var byUnit = {};
  for (var i=1; i<refData.length; i++) {
    var g = String(refData[i][0]).trim(), s = String(refData[i][1]).trim(), sub = String(refData[i][2]).trim(), unit = String(refData[i][3]).trim();
    if (!unit) continue;
    if (filter) {
      if (filter.grade && filter.grade !== g) continue;
      if (filter.semester && filter.semester !== s) continue;
      if (filter.subject && filter.subject !== sub) continue;
    }
    var key = g+'|'+s+'|'+sub+'|'+unit;
    if (!byUnit[key]) byUnit[key] = { grade: g, semester: s, subject: sub, unit: unit, refs: [] };
    byUnit[key].refs.push({ text: refData[i][5] || '', goal: refData[i][12] || '' });
  }
  return { keys: Object.keys(byUnit), byUnit: byUnit };
}

// ========== Firebase 토큰 발급 함수 ==========
function getToken_() {
  try {
    var serviceAccountJson = PropertiesService.getScriptProperties().getProperty('FIREBASE_SERVICE_ACCOUNT');
    if (!serviceAccountJson) {
      log_('getToken_', '❌ FIREBASE_SERVICE_ACCOUNT 속성 없음');
      return null;
    }
    var serviceAccount = JSON.parse(serviceAccountJson);
    var jwtHeader = Utilities.base64EncodeWebSafe(JSON.stringify({ alg: 'RS256', typ: 'JWT' }));
    var now = Math.floor(Date.now() / 1000);
    var jwtClaim = Utilities.base64EncodeWebSafe(JSON.stringify({
      iss: serviceAccount.client_email,
      scope: 'https://www.googleapis.com/auth/datastore',
      aud: 'https://oauth2.googleapis.com/token',
      iat: now,
      exp: now + 3600
    }));
    var signatureInput = jwtHeader + '.' + jwtClaim;
    var signature = Utilities.base64EncodeWebSafe(Utilities.computeRsaSha256Signature(signatureInput, serviceAccount.private_key));
    var jwt = signatureInput + '.' + signature;
    var response = UrlFetchApp.fetch('https://oauth2.googleapis.com/token', {
      method: 'post',
      payload: 'grant_type=urn:ietf:params:oauth:grant-type:jwt-bearer&assertion=' + jwt,
      muteHttpExceptions: true
    });
    var result = JSON.parse(response.getContentText());
    if (result.access_token) {
      return result.access_token;
    } else {
      log_('getToken_', '토큰 응답 오류: ' + JSON.stringify(result));
      return null;
    }
  } catch (e) {
    log_('getToken_', '예외 발생: ' + e.message);
    return null;
  }
}

function buildGenPrompt_(info, gradeNum, numRequired) {
  var example = (info.refs && info.refs[0]) ? info.refs[0].text : '';
  var goal = (info.refs && info.refs[0]) ? info.refs[0].goal : '';
  var dynamicRules = [];
  try { dynamicRules = generateDynamicRules(); } catch(e) {}
  var ruleText = dynamicRules.length ? "\n\n[자동 생성된 개선 규칙 - 반드시 준수]\n" + dynamicRules.join("\n") : "";
  return `당신은 초등 ${gradeNum}학년 ${info.subject} 교과서 집필 위원입니다.
아래 단원과 참고 문제 예시를 분석하여 완전히 새로운 문제를 만드세요.
[단원] ${info.unit}
[핵심 개념] ${goal || '해당 단원의 핵심 개념'}
[참고 예시] ${example}
[생성 개수] 반드시 ${numRequired}개의 문제를 JSON 배열로 출력하세요.

[JSON 규칙 - 반드시 준수]
1. 5지선다, 정답 인덱스(0~4)는 논리적 정답에 맞게 지정.
2. options는 5개의 문자열 배열.
3. visual_type: "emoji_scene" 사용. 예: {"items":[{"emoji":"😊","count":1,"label":"웃음"}],"total":1,"layout":"grid"}
4. 해설에 정답 이유와 각 오답에 대한 간단한 이유를 포함.
5. 난이도: "easy", "medium", "hard".
6. 해설에서 '정답은 X번'이라 쓸 때, X는 보기 순서(1-indexed)를 따릅니다.
${ruleText}
[JSON 예시]
[{"question":"질문","options":["보기1","보기2","보기3","보기4","보기5"],"answer":0,"explanation":"정답은 1번입니다. 이유는...","difficulty":"easy","visual_type":"emoji_scene","visual_data":{"items":[{"emoji":"😊","count":1,"label":"웃음"}],"total":1,"layout":"grid"}}]

시작:
`;
}

function buildDeepSeekVerifyPrompt_(problem, gradeNum) {
  return `당신은 초등 ${gradeNum}학년 교육 전문가입니다. 아래 문제를 평가하세요.

[문제] ${JSON.stringify(problem, null, 2)}

[검증 항목]
1. 정답 유일성
2. 보기 일관성
3. 해설 충실도 (각 오답 이유 포함?)
4. 시각자료 적절성 (이모지 사용 권장, 빈 객체 금지)
5. 난이도 적합성
6. answer 필드 일치 (0 고정 금지)

[추가 검증 - 해설과 answer 일치]
- 해설에 "정답은 X번"이 있다면, X == answer+1 이어야 함.
- 예: answer=0 → 해설에 "정답은 1번" (PASS), "정답은 0번" (CORRECTED)

출력 JSON: {"verdict":"PASS/CORRECTED/FAIL","score":0~100,"feedback":"구체적 지적"}
`;
}

function buildGeminiVerifyPrompt_(problem, gradeNum) {
  return `JSON 형식 검증: ${JSON.stringify(problem, null, 2)}

[추가 검증 - answer와 해설 일치]
- 해설에 '정답은 X번' 형태가 있다면, X == answer+1 이어야 함.

출력: {"verdict":"PASS/FAIL/CORRECTED","score":0~100,"feedback":"..."}
`;
}

function fixWithDeepSeek_(original, feedback, gradeNum) {
  return `다음 문제를 피드백에 따라 수정하세요. JSON만 출력.
원본: ${JSON.stringify(original)}
피드백: ${feedback}
`;
}

function step1_RunWithFilter(filter) {
  var props = PropertiesService.getScriptProperties();
  if (filter.regenerate) {
    var deleted = deleteRangeFromGenSheet_(filter);
    log_('Step1', '🗑️ 재생성: ' + deleted + '행 삭제');
  }
  var subjects = filter.subjects || (filter.subject ? [filter.subject] : []);
  if (subjects.length === 0) { log_('Step1', '⚠️ 과목 미선택'); return; }
  var selectedSubject = subjects[0];
  var singleFilter = { grade: filter.grade, semester: filter.semester, subject: selectedSubject, target: filter.target || 5, append: filter.append !== false, regenerate: filter.regenerate === true };
  props.setProperty('current_filter', JSON.stringify(singleFilter));
  props.setProperty('gen_resume', '0');
  runGenerate_(false, singleFilter);
}

function step1_Resume() {
  deleteTriggers_('step1_Resume');
  var props = PropertiesService.getScriptProperties();
  var resumeFilterJson = props.getProperty('gen_resume_filter') || props.getProperty('current_filter');
  if (!resumeFilterJson) { log_('Step1_Resume', '❌ 재개 필터 없음'); return; }
  var filter = JSON.parse(resumeFilterJson);
  log_('Step1_Resume', '▶️ 재개: ' + filter.grade + '학년 ' + filter.semester + '학기 ' + filter.subject);
  runGenerate_(false, filter);
}

function autoVerifyAfterGen() {
  deleteTriggers_('autoVerifyAfterGen');
  var props = PropertiesService.getScriptProperties();
  var pendingQids = props.getProperty('pending_verify_qids');
  if (pendingQids) {
    var qids = JSON.parse(pendingQids);
    log_('AutoVerify', '🔄 예약된 자동 검증 실행 (' + qids.length + '개 문제)');
    runVerify_(false);
    props.deleteProperty('pending_verify_qids');
  }
}

function runGenerate_(isNew, filter) {
  log_('Step1', '🚀 문제 생성 시작 at ' + new Date().toLocaleString() + ' 필터: ' + JSON.stringify(filter));
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var props = PropertiesService.getScriptProperties();
  var data = getUnitKeys_(filter || {});
  var keys = data.keys, byUnit = data.byUnit;
  if (keys.length === 0) { log_('Step1', '⚠️ 조건 맞는 참고문제 없음'); return { total: 0, paused: false }; }

  var target = (filter && filter.target) ? filter.target : 99999;
  var genSheet = ss.getSheetByName('생성문제');
  if (!genSheet) { genSheet = ss.insertSheet('생성문제'); initGenSheet_(genSheet); }

  var row = genSheet.getLastRow() + 1;
  var resumeIdx = parseInt(props.getProperty('gen_resume') || '0');
  var start = Date.now();
  var total = 0, dupSkip = 0, unitSkip = 0, visualDemoted = 0, emptyRetry = 0;
  var skippedUnits = [];

  for (var k = resumeIdx; k < keys.length; k++) {
    if (Date.now() - start > TIME_LIMIT) {
      props.setProperty('gen_resume', String(k));
      props.setProperty('gen_resume_filter', JSON.stringify(filter));
      deleteTriggers_('step1_Resume');
      ScriptApp.newTrigger('step1_Resume').timeBased().after(70 * 1000).create();
      log_('Step1', '⏰ 시간 초과로 중단, ' + total + '개 생성됨');
      if (total > 0) {
        var newQids = [];
        var startRow = row - total;
        for (var r = startRow; r < row; r++) { var qid = genSheet.getRange(r, 17).getValue(); if (qid) newQids.push(qid); }
        if (newQids.length > 0) { props.setProperty('pending_verify_qids', JSON.stringify(newQids)); deleteTriggers_('autoVerifyAfterGen'); ScriptApp.newTrigger('autoVerifyAfterGen').timeBased().after(1000).create(); }
      }
      return { total: total, paused: true };
    }

    var info = byUnit[keys[k]];
    var gradeNum = parseInt(info.grade) || 1;
    var unitQ = collectUnitQuestions_(genSheet, info);

    try {
      var allProblems = [];
      var need = target;
      var maxAttempts = 5;
      for (var attempt = 0; attempt < maxAttempts && allProblems.length < need; attempt++) {
        var prompt = buildGenPrompt_(info, gradeNum, need - allProblems.length);
        var raw = callGemini_(prompt, 0.7, 8000);
        var result = parseJSON_(raw);
        if (!result) { Utilities.sleep(3000); continue; }
        if (!Array.isArray(result)) result = [result];
        for (var q = 0; q < result.length && allProblems.length < need; q++) {
          var qu = result[q];
          var dsRes = callDeepSeek_(buildDeepSeekVerifyPrompt_(qu, gradeNum), 2000);
          var dsJson = parseJSON_(dsRes);
          if (!dsJson || dsJson.verdict === 'FAIL') continue;
          if (dsJson.verdict === 'CORRECTED') {
            var fixedRaw = callDeepSeek_(fixWithDeepSeek_(qu, dsJson.feedback, gradeNum), 4000);
            var fixed = parseJSON_(fixedRaw);
            if (fixed) qu = fixed;
          }
          var gemRes = callGemini_(buildGeminiVerifyPrompt_(qu, gradeNum), 0.2, 1500);
          var gemJson = parseJSON_(gemRes);
          if (!gemJson || gemJson.verdict !== 'PASS') continue;
          var dup = findDuplicate_(qu.question || '', unitQ.concat(allProblems.map(function(p){return p.question;})));
          if (dup) continue;
          allProblems.push(qu);
        }
        Utilities.sleep(2000);
      }
      if (allProblems.length === 0) { log_('Step1', '⚠️ ' + info.unit + ' 생성 실패'); continue; }
      var now = new Date();
      var ts = Utilities.formatDate(now, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
      for (var p = 0; p < allProblems.length; p++) {
        var qu = allProblems[p];
        var opts = qu.options || [];
        while (opts.length < 5) opts.push('');
        opts = opts.slice(0,5);
        var ansIdx = 0;
        if (qu.answer !== undefined && qu.answer !== null) {
          var rawAns = qu.answer;
          if (typeof rawAns === 'string' && !isNaN(rawAns)) ansIdx = parseInt(rawAns,10);
          else if (typeof rawAns === 'number') ansIdx = rawAns;
          else if (typeof rawAns === 'string' && isNaN(rawAns)) { var idx = opts.indexOf(rawAns); ansIdx = (idx >=0) ? idx : 0; }
        }
        if (isNaN(ansIdx) || ansIdx<0 || ansIdx>=5) ansIdx = 0;
        var explanation = qu.explanation || '';
        explanation = explanation.replace(/정답은 \d+번/gi, '정답은 ' + (ansIdx+1) + '번');
        explanation = explanation.replace(/정답은 -1번/gi, '정답은 ' + (ansIdx+1) + '번');
        qu.explanation = explanation;
        var visualType = qu.visual_type || 'emoji_scene';
        var visualData = qu.visual_data;
        if (visualType === 'emoji_scene' && (!visualData || Object.keys(visualData).length===0)) { visualData = { items: [{ emoji: "📚", count: 1, label: "학습" }], total: 1, layout: "grid" }; }
        if (visualType === 'text_only') visualData = null;
        var vCheck = validateVisual_(visualType, visualData || {});
        if (!vCheck.ok) { log_('Step1', '⚠️ visual 검증 실패 → text_only 강등: ' + vCheck.reason); visualType = 'text_only'; visualData = null; visualDemoted++; }
        var qid = hashId_(info.grade+'|'+info.semester+'|'+info.subject+'|'+info.unit+'|'+qu.question);
        genSheet.getRange(row,1,1,17).setValues([[
          info.grade, info.semester, info.subject, info.unit,
          qu.difficulty || 'medium', qu.question || '',
          clean_(opts[0]), clean_(opts[1]), clean_(opts[2]), clean_(opts[3]), clean_(opts[4]),
          ansIdx, qu.explanation || '',
          visualType, visualData ? JSON.stringify(visualData) : '',
          ts, qid
        ]]);
        row++; total++; unitQ.push(qu.question);
      }
      SpreadsheetApp.flush();
    } catch(e) { log_('Step1', '❌ ' + info.unit + ': ' + e.message); }
  }

  props.deleteProperty('gen_resume'); props.deleteProperty('gen_resume_filter');
  deleteTriggers_('step1_Resume');
  log_('Step1', '🏁 문제 생성 종료 at ' + new Date().toLocaleString() + ' 총 생성: ' + total + '개');
  if (total > 0) {
    log_('Step1', '🔄 자동 검증 시작 (' + total + '개 문제)');
    try { var autoVSheet = ss.getSheetByName('검증결과'); if (!autoVSheet) { autoVSheet = ss.insertSheet('검증결과'); initVerifySheet_(autoVSheet); } props.setProperty('verify_resume','0'); runVerify_(false); } catch(e) { log_('Step1', '⚠️ 자동 검증 실패: ' + e.message); }
  }
  return { total: total, paused: false, skippedUnits: skippedUnits };
}

function normalizeVisualData(visualType, data) { if (visualType !== 'emoji_scene' || !data) return data; var norm = { items: [], total: data.total || 0, layout: data.layout || 'grid' }; if (Array.isArray(data.items)) { for (var i=0; i<data.items.length; i++) { var item = data.items[i]; if (typeof item === 'string') norm.items.push({ emoji: item, count:1, label: item }); else if (typeof item === 'object' && item.emoji) norm.items.push(item); } } else if (typeof data.items === 'string') { var emojiMap = {}; var chars = data.items.split(''); chars.forEach(function(ch) { emojiMap[ch] = (emojiMap[ch]||0)+1; }); for (var em in emojiMap) norm.items.push({ emoji: em, count: emojiMap[em], label: em }); } if (!norm.total && norm.items.length) { norm.total = norm.items.reduce(function(s,it){ return s + (it.count||1); },0); } return norm; }

function runVerify_(isManual) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var genSheet = ss.getSheetByName('생성문제');
  if (!genSheet || genSheet.getLastRow() <= 1) return;
  var verifySheet = ss.getSheetByName('검증결과');
  if (!verifySheet) { verifySheet = ss.insertSheet('검증결과'); initVerifySheet_(verifySheet); }

  var genData = genSheet.getDataRange().getValues();
  var existingQids = getExistingQids_(verifySheet);
  var processed = 0;
  for (var i = 1; i < genData.length; i++) {
    var row = genData[i];
    var qid = row[16];
    if (existingQids[qid]) continue;
    var problem = {
      grade: row[0], semester: row[1], subject: row[2], unit: row[3],
      difficulty: row[4], question: row[5],
      options: [row[6], row[7], row[8], row[9], row[10]],
      answer: row[11], explanation: row[12],
      visual_type: row[13], visual_data: row[14]
    };
    // DeepSeek 검증
    var dsRes = callDeepSeek_(buildDeepSeekVerifyPrompt_(problem, parseInt(problem.grade)), 2000);
    var dsJson = parseJSON_(dsRes);
    var dsVerdict = dsJson ? dsJson.verdict : 'FAIL';
    var dsScore = dsJson ? dsJson.score : 0;
    var dsFeedback = dsJson ? dsJson.feedback : '';
    // Gemini 검증
    var gemRes = callGemini_(buildGeminiVerifyPrompt_(problem, parseInt(problem.grade)), 0.2, 1500);
    var gemJson = parseJSON_(gemRes);
    var gemVerdict = gemJson ? gemJson.verdict : 'FAIL';
    var gemScore = gemJson ? gemJson.score : 0;
    var gemFeedback = gemJson ? gemJson.feedback : '';
    // 최종 판정
    var finalVerdict = 'FAIL';
    if (dsVerdict === 'PASS' && gemVerdict === 'PASS') finalVerdict = 'PASS';
    else if (dsVerdict === 'CORRECTED' || gemVerdict === 'CORRECTED') finalVerdict = 'CORRECTED';
    else if (dsVerdict === 'PASS' && gemVerdict === 'FAIL') finalVerdict = 'CORRECTED';
    // 저장
    verifySheet.appendRow([
      problem.grade, problem.semester, problem.subject, problem.unit,
      problem.difficulty, problem.question,
      problem.options[0], problem.options[1], problem.options[2], problem.options[3], problem.options[4],
      problem.answer, problem.explanation,
      problem.visual_type, problem.visual_data,
      new Date().toLocaleString(), qid,
      dsVerdict, dsScore, dsFeedback,
      gemVerdict, gemScore, gemFeedback,
      finalVerdict
    ]);
    processed++;
    Utilities.sleep(1000);
  }
  log_('Verify', '✅ 검증 완료: ' + processed + '개');
  if (isManual) SpreadsheetApp.getUi().alert('✅ 검증 완료: ' + processed + '개');
  try { collectAndSaveFeedback_(); } catch(e) { log_('Verify', '피드백 저장 실패: '+e.message); }
}

function getExistingQids_(sheet) {
  var data = sheet.getDataRange().getValues();
  var qids = {};
  for (var i=1; i<data.length; i++) { var q = data[i][16]; if(q) qids[q]=true; }
  return qids;
}

function collectAndSaveFeedback_() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var verifySheet = ss.getSheetByName('검증결과');
  var fbSheet = ss.getSheetByName('피드백은행');
  if (!verifySheet || verifySheet.getLastRow() <= 1) return;
  if (!fbSheet) { fbSheet = ss.insertSheet('피드백은행'); initFeedbackBank_(fbSheet); }
  var vData = verifySheet.getDataRange().getValues();
  var existing = getExistingFeedbacks_(fbSheet);
  for (var i = 1; i < vData.length; i++) {
    var finalVerdict = String(vData[i][23] || '').trim();
    if (finalVerdict !== 'CORRECTED' && finalVerdict !== 'FAIL') continue;
    var subject = vData[i][2], unit = vData[i][3];
    var fb = String(vData[i][19] || '') + ' ' + String(vData[i][22] || '');
    if (!fb.trim()) continue;
    var errorCode = classifyFeedback_(fb);
    if (errorCode === 'E000') continue;
    if (isDuplicateFeedback_(subject, unit, errorCode, fb, existing)) continue;
    fbSheet.appendRow([subject, unit, errorCode, fb, '검증결과', new Date().toLocaleString(), 'N']);
    existing.push({ subject: subject, unit: unit, errorCode: errorCode, content: fb });
  }
}

function getExistingFeedbacks_(sheet) {
  var data = sheet.getDataRange().getValues();
  var list = [];
  for (var i = 1; i < data.length; i++) list.push({ subject: data[i][0], unit: data[i][1], errorCode: data[i][2], content: data[i][3] });
  return list;
}

function isDuplicateFeedback_(subject, unit, errorCode, newContent, existingList) {
  var norm = newContent.replace(/\s+/g, ' ').toLowerCase();
  for (var i = 0; i < existingList.length; i++) {
    if (existingList[i].subject === subject && existingList[i].unit === unit && existingList[i].errorCode === errorCode) {
      var existingNorm = existingList[i].content.replace(/\s+/g, ' ').toLowerCase();
      if (jaccardSim_(norm, existingNorm) > 0.8) return true;
    }
  }
  return false;
}

function classifyFeedback_(text) {
  var t = text.toLowerCase();
  if (t.includes('visual') || t.includes('스키마') || t.includes('빈 객체') || t.includes('시각자료') || t.includes('이모지') && (t.includes('적절') || t.includes('부족'))) return 'E006';
  if (t.includes('보기') || t.includes('일관성') || t.includes('이질적') || t.includes('중복') || t.includes('범주')) return 'E002';
  if (t.includes('해설') || t.includes('설명') || t.includes('반박') || t.includes('누락')) return 'E004';
  if (t.includes('지문') || t.includes('그림 설명') || t.includes('누락')) return 'E003';
  if (t.includes('정답') || t.includes('인덱스') || t.includes('유일성')) return 'E001';
  if (t.includes('문맥') || t.includes('충돌')) return 'E005';
  if (t.includes('이모지') && (t.includes('일치') || t.includes('허용'))) return 'E007';
  return 'E000';
}

function getQualifiedFeedback_(subject, unit) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('피드백은행');
  if (!sheet || sheet.getLastRow() <= 1) return [];
  var data = sheet.getDataRange().getValues();
  var now = new Date();
  var result = [];
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    if (row[0] !== subject || row[1] !== unit) continue;
    if (row[6] === 'Y') continue;
    var dateCreated = new Date(row[5]);
    if ((now - dateCreated) > 30 * 24 * 3600 * 1000) continue;
    result.push({ code: row[2], text: row[3] });
  }
  return result.slice(0, 10);
}

var SUBJECT_EN = {
  '국어': 'korean', '수학': 'math', '통합교과': 'integrated',
  '과학': 'science', '사회': 'social', '영어': 'english',
};

var DIFFICULTY_EN = {
  '상': 'hard', '중': 'medium', '하': 'easy',
  '상급': 'hard', '중급': 'medium', '하급': 'easy',
};

// === Firestore 필드 변환 ===
function objToFirestoreFields(obj) {
  var fields = {};
  for (var key in obj) {
    var val = obj[key];
    if (val === null || val === undefined) {
      fields[key] = { nullValue: null };
    } else if (typeof val === 'string') {
      fields[key] = { stringValue: val };
    } else if (typeof val === 'number') {
      fields[key] = Number.isInteger(val) ? { integerValue: val } : { doubleValue: val };
    } else if (typeof val === 'boolean') {
      fields[key] = { booleanValue: val };
    } else if (Array.isArray(val)) {
      fields[key] = {
        arrayValue: {
          values: val.map(function (v) {
            if (typeof v === 'string') return { stringValue: v };
            if (typeof v === 'number') return Number.isInteger(v) ? { integerValue: v } : { doubleValue: v };
            if (typeof v === 'boolean') return { booleanValue: v };
            if (v === null) return { nullValue: null };
            if (typeof v === 'object') return { mapValue: { fields: objToFirestoreFields(v) } };
            return { stringValue: String(v) };
          })
        }
      };
    } else if (typeof val === 'object') {
      fields[key] = { mapValue: { fields: objToFirestoreFields(val) } };
    }
  }
  return fields;
}

// === 행 → Firestore 문서 변환 ===
function rowToAppDoc(row) {
  var questionId = String(row[16] || '').trim() || Utilities.getUuid();
  var subjectKr = String(row[2] || '').trim();
  var subjectEn = SUBJECT_EN[subjectKr] || subjectKr.toLowerCase();

  var options = [];
  for (var c = 6; c <= 10; c++) {
    var opt = String(row[c] || '').trim();
    if (opt) options.push(opt);
  }

  var rawAnswer = String(row[11] !== null && row[11] !== undefined ? row[11] : '').trim();

  var answer;
  var answerIdx = options.indexOf(rawAnswer);
  if (answerIdx >= 0) {
    answer = answerIdx;
  } else {
    answer = rawAnswer;
  }

  var visualDataRaw = String(row[14] || '').trim();
  var visualData = null;
  if (visualDataRaw) {
    try { visualData = JSON.parse(visualDataRaw); } catch (e) { visualData = null; }
  }

  var difficultyRaw = String(row[4] || '').trim();
  var difficulty = DIFFICULTY_EN[difficultyRaw] || 'medium';

  var fields = {};

  var stringFields = {
    question: String(row[5] || ''),
    explanation: String(row[12] || ''),
    unit: String(row[3] || ''),
    subject: subjectEn,
    difficulty: difficulty,
    type: 'multiple_choice',
    visual_type: String(row[13] || '').trim() || null,
    createdAt: String(row[15] || new Date().toISOString()),
    questionId: questionId,
    claudeVerdict: String(row[17] || ''),
    claudeFeedback: String(row[19] || ''),
    geminiVerdict: String(row[20] || ''),
    geminiFeedback: String(row[22] || ''),
    finalVerdict: String(row[23] || ''),
  };

  for (var sk in stringFields) {
    var sv = stringFields[sk];
    if (sv !== null && sv !== undefined && sv !== '') {
      fields[sk] = { stringValue: String(sv) };
    } else if (sv === null && sk === 'visual_type') {
      fields[sk] = { nullValue: null };
    }
  }

  fields['grade'] = { integerValue: parseInt(String(row[0] || '1')) || 1 };
  fields['semester'] = { integerValue: parseInt(String(row[1] || '1')) || 1 };
  fields['claudeScore'] = { integerValue: parseInt(String(row[18] || '0')) || 0 };
  fields['geminiScore'] = { integerValue: parseInt(String(row[21] || '0')) || 0 };
  fields['isActive'] = { booleanValue: true };

  if (options.length > 0) {
    fields['options'] = {
      arrayValue: { values: options.map(function (o) { return { stringValue: o }; }) }
    };
  }

  if (typeof answer === 'number') {
    fields['answer'] = { integerValue: answer };
  } else {
    fields['answer'] = { stringValue: String(answer) };
  }

  if (visualData && typeof visualData === 'object' && !Array.isArray(visualData)) {
    fields['visual_data'] = { mapValue: { fields: objToFirestoreFields(visualData) } };
  } else {
    fields['visual_data'] = { nullValue: null };
  }

  return { questionId: questionId, fields: fields };
}

// === Step3 수동 업로드 ===
function step3_Upload() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var resultSheet = ss.getSheetByName('검증결과');
  if (!resultSheet) {
    ui.alert('❌ 검증결과 시트가 없습니다.');
    return;
  }

  var data = resultSheet.getDataRange().getValues();
  if (data.length <= 1) {
    ui.alert('❌ 업로드할 데이터가 없습니다.');
    return;
  }

  var finalCol = 23;

  var uploadRows = [];
  for (var i = 1; i < data.length; i++) {
    var verdict = String(data[i][finalCol] || '').trim().toUpperCase();
    if (verdict === 'PASS' || verdict === 'CORRECTED') {
      uploadRows.push({ row: data[i], idx: i });
    }
  }

  var seen = {};
  var finalRows = [];
  for (var k = 0; k < uploadRows.length; k++) {
    var qid = String(uploadRows[k].row[16] || '').trim();
    if (!qid || seen[qid]) continue;
    seen[qid] = true;
    finalRows.push(uploadRows[k]);
  }

  if (finalRows.length === 0) {
    ui.alert('❌ PASS 또는 CORRECTED 항목이 없습니다.');
    return;
  }

  var answer = ui.alert(
    '📤 Firestore 업로드',
    '총 ' + finalRows.length + '개 문제를 업로드합니다.\n\n계속하시겠습니까?',
    ui.ButtonSet.YES_NO
  );
  if (answer !== ui.Button.YES) return;

  executeUpload_(finalRows);
}

// === 실제 업로드 실행 (공통) ===
function executeUpload_(finalRows) {
  var token = getToken_();
  if (!token) {
    log_('업로드', '❌ 토큰 발급 실패');
    try { SpreadsheetApp.getUi().alert('❌ 토큰 발급 실패'); } catch (e) {}
    return;
  }

  var writes = [];
  for (var w = 0; w < finalRows.length; w++) {
    var doc = rowToAppDoc(finalRows[w].row);
    writes.push({
      update: {
        name: 'projects/' + PROJECT_ID + '/databases/(default)/documents/questions/' + doc.questionId,
        fields: doc.fields
      }
    });
  }

  var baseUrl = 'https://firestore.googleapis.com/v1/projects/' + PROJECT_ID + '/databases/(default)/documents:batchWrite';
  var BATCH_LIMIT = 500;
  var successCount = 0, failCount = 0;

  for (var b = 0; b < writes.length; b += BATCH_LIMIT) {
    var batch = writes.slice(b, b + BATCH_LIMIT);
    var payload = JSON.stringify({ writes: batch });

    try {
      var response = UrlFetchApp.fetch(baseUrl, {
        method: 'post',
        contentType: 'application/json',
        headers: { Authorization: 'Bearer ' + token },
        payload: payload,
        muteHttpExceptions: true
      });
      var code = response.getResponseCode();

      if (code === 200) {
        var json = JSON.parse(response.getContentText());
        var statuses = json.status || [];
        for (var s = 0; s < statuses.length; s++) {
          if (!statuses[s] || !statuses[s].code || statuses[s].code === 0) {
            successCount++;
          } else {
            failCount++;
            log_('업로드', '❌ 문서 실패: ' + JSON.stringify(statuses[s]));
          }
        }
      } else {
        failCount += batch.length;
        log_('업로드', '❌ HTTP ' + code + ': ' + response.getContentText().substring(0, 300));
      }
    } catch (e) {
      failCount += batch.length;
      log_('업로드', '❌ 예외: ' + e.message);
    }

    if (b + BATCH_LIMIT < writes.length) Utilities.sleep(1000);
  }

  log_('업로드', '✅ 성공:' + successCount + ' 실패:' + failCount);
  try {
    SpreadsheetApp.getUi().alert('🎉 업로드 완료!\n\n성공: ' + successCount + '개\n실패: ' + failCount + '개');
  } catch (e) {}
}

function diagnoseUpload_() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var results = [];

  var vs = ss.getSheetByName('검증결과');
  if (!vs) {
    results.push('❌ 검증결과 시트가 없습니다.');
  } else {
    var rows = vs.getLastRow() - 1;
    results.push('✅ 검증결과 시트 있음: ' + rows + '행');

    var header = vs.getRange(1, 1, 1, vs.getLastColumn()).getValues()[0];
    results.push('📋 컬럼 개수: ' + header.length);

    var finalCol = -1;
    for (var h = 0; h < header.length; h++) {
      if (String(header[h]).indexOf('최종판정') >= 0) finalCol = h;
    }
    results.push(finalCol >= 0 ? '✅ 최종판정 컬럼 위치: ' + finalCol : '❌ 최종판정 컬럼을 찾을 수 없음');

    if (rows > 0 && finalCol >= 0) {
      var data = vs.getDataRange().getValues();
      var p = 0, c = 0;
      for (var i = 1; i < data.length; i++) {
        var v = String(data[i][finalCol] || '').trim();
        if (v === 'PASS') p++;
        else if (v === 'CORRECTED') c++;
      }
      results.push('📊 PASS: ' + p + ' / CORRECTED: ' + c + ' = 총 ' + (p + c) + '개 업로드 대상');
    }
  }

  try {
    var raw = PropertiesService.getScriptProperties().getProperty('FIREBASE_SERVICE_ACCOUNT');
    if (!raw) {
      results.push('❌ FIREBASE_SERVICE_ACCOUNT 속성이 없습니다.');
    } else {
      var k = JSON.parse(raw);
      results.push('✅ 서비스 계정 있음: ' + k.client_email);
    }
  } catch (e) {
    results.push('❌ 서비스 계정 JSON 파싱 실패: ' + e.message);
  }

  try {
    var tok = getToken_();
    if (tok) {
      results.push('✅ JWT 토큰 발급 성공 (길이: ' + tok.length + ')');

      var testUrl = 'https://firestore.googleapis.com/v1/projects/' + PROJECT_ID + '/databases/(default)/documents/test_conn';

        method: 'get',
        headers: { 'Authorization': 'Bearer ' + tok },
        muteHttpExceptions: true
      });
      var code = r2.getResponseCode();
      results.push('🔗 Firestore Bearer 테스트: ' + code);
      if (code !== 200) results.push('   응답: ' + r2.getContentText().substring(0, 200));
    } else {
      results.push('❌ 토큰 발급 실패 (null 반환)');
    }
  } catch (e) {
    results.push('❌ 토큰/연결 테스트 오류: ' + e.message);
  }

  ui.alert('🔍 업로드 진단 결과\n\n' + results.join('\n'));
